001    package edu.mccc.cos210.cmwar;
002    import java.awt.Point;
003    import java.util.List;
004    
005    /**
006    * Component interface for Circuit Maker with AutoRouter.
007    * A Component is an item that can be used in a circuit
008    * @author Rohit Kumar
009    * @version 0.1 
010    */
011    public interface Component {
012            /**
013            * Connects this component to another
014            * @param c Component to connect to.
015            */
016            public void connectTo(Component c);
017    
018            /*
019            * Rotates this component
020            */
021            public void rotate();
022    
023            /**
024            * Returns the orientation of the component (changed with rotate())
025            * @return Orientation enum describing current orientation: Up, Down, Left, Right (facing)
026            */
027            public Orientation getOrientation();
028    
029            /**
030            * Returns the name of this component.
031            * Likely helpful for use in GUI, and debugging
032            * @return String name of component 
033            */
034            public String getName();
035    
036            /**
037            * Returns a list of Connections this component has
038            * @return List of connections component has
039            */
040            public List<Connection> getConnections();
041    
042            /**
043            * Returns location of Component
044            * @return Point object with location coordinates
045            */
046            public Point getPosition();
047            
048            /**
049            * Sets the position
050            * @param x X coordinate for new location of component
051            * @param y Y coordination for new location of component
052            */
053            public void setPosition(int x, int y);
054    
055    }
