001    package edu.mccc.cos210.cmwar;
002    import java.awt.Rectangle;
003    import java.awt.Point;
004    import java.awt.Dimension;
005    
006    /**
007    * ConnectionPoint: Specifies individual areas on components that can be connected.
008    * @author Rohit Kumar
009    * @version 0.1
010    */
011    public interface ConnectionPoint {
012            /**
013            *Returns the bounds (size) of connection point
014            *@return Dimension: size of bounds
015            */
016            public Dimension getBounds();
017    
018            /**
019            *Sets bounds size of connection point
020            *@param width width of bounding box
021            *@param height height of bounding box
022            */
023            public void setBounds(int width, int height);
024    
025            /**
026            *Gets location of bounding box
027            *@return Location of bounding box.
028            */
029            public Point getLocation();
030    
031            /**
032            *Sets location of bounding box
033            */
034            public void setLocation(int x, int y);
035    }
