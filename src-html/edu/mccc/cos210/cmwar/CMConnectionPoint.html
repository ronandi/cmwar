001    package edu.mccc.cos210.cmwar;
002    import java.awt.Rectangle;
003    import java.awt.Point;
004    import java.awt.Dimension;
005    
006    /**
007    * CMConnectionPoint: Specifies individual areas on components that can be connected.
008    * @author Rohit Kumar
009    * @version 0.1
010    * 
011    */
012    public class CMConnectionPoint implements ConnectionPoint {
013            
014            /**
015            *Bounding box of a connection point.
016            */
017            private Rectangle myBounds;
018            public CMConnectionPoint(int x, int y, int width, int height) {
019                    myBounds = new Rectangle(x, y, width, height);
020            }
021    
022            /**
023            *Sets bounds size of connection point
024            *@test.status ok
025            *@param width width of bounding box
026            *@param height height of bounding box
027            */
028            public void setBounds(int width, int height) {
029                    myBounds.setSize(width, height);
030            }
031    
032            /**
033            *Returns the bounds (size) of connection point
034            *@test.status ok
035            *@return Dimension: size of bounds
036            */
037            public Dimension getBounds() {
038                    return myBounds.getSize();
039            }
040    
041            /**
042            *Gets location of bounding box.
043            *@test.status ok
044            *@return Location of bounding box.
045            */
046            public Point getLocation() {
047                    return myBounds.getLocation();
048            }
049    
050            /**
051            *Sets location of bounding box
052            *@test.status ok
053            */
054            public void setLocation(int x, int y) {
055                    myBounds.setLocation(x, y);
056            }
057            public static void main(String[] sa) throws Exception {
058                    System.out.println("CMConnectionPoint.java test");
059                    ConnectionPoint cmcp = new CMConnectionPoint(1, 3, 5, 5);
060                    System.out.println("5x5 connection point at " + cmcp.getLocation() + " created");
061                    System.out.println(cmcp.getBounds());
062                    cmcp.setBounds(10, 10);
063                    System.out.println(cmcp.getBounds());
064                    cmcp.setLocation(2, 4);
065                    System.out.println(cmcp.getLocation());
066                    System.out.println("Test completed");
067            }
068    
069    }
