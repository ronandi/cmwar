001    package edu.mccc.cos210.cmwar;
002    import java.util.List;
003    /** 
004    * Board interface for Circuit Maker with AutoRoute
005    * A board has Components and Connections, and manages them. 
006    * @author Rohit Kumar
007    * @version 0.1
008    */
009    public interface Board {
010            /**
011            * Adds a component to the board.
012            * @param c A valid component to be added to the board.
013            */
014            public void addComponent(Component c);
015            
016            /**
017            * Deletes a component on the board.
018            * @param c A valid component already on the board.
019            */
020            public void deleteComponent(Component c);
021            
022            /**
023            * Returns a list of all the components on the board.
024            * @return a List of all the Components already on the board.
025            */
026            public List<Component> listComponents();
027            
028            /**
029            * Returns a list of all the connections between components on the board.
030            * @return Returns a list of all the connections between components on the board.
031            */
032            public List<Connection> listConnections();
033    
034            /**
035            * Adds a connection between two components 
036            * @param a A Component to be connected.
037            * @param b Opposite Component to be connected to A, and vice versa.
038            */
039            public void addConnection(Component a, Component b);
040    
041            /** 
042            * The long-dreaded method that does all the big work. Transforms the rat's nest into traces.
043            */
044            public void autoRoute();
045            
046            /**
047            * Does the opposite of autoRoute(). Returns the traced board to a rat's nest.
048            */
049            public void ripUp();
050    
051            /**
052            * Deletes all connections between componets on the board.
053            */
054            public void clearConnections();
055            
056            /**
057            * Saves the board in its current state.
058            */
059            public void save();
060            
061            /**
062            * Loads the board in the state it was saved in.
063            */
064            public void load();
065    }
