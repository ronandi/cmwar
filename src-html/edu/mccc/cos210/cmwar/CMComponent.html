001    package edu.mccc.cos210.cmwar;
002    import java.awt.Point;
003    import java.util.List;
004    import java.util.ArrayList;
005    
006    /**
007    * Component interface for Circuit Maker with AutoRouter.
008    * A Component is an item that can be used in a circuit
009    * @author Rohit Kumar
010    * @version 0.1 
011    */
012    public class CMComponent implements Component {
013            /**
014            *List of ConnectionPoints a Component has.
015            */
016            private ArrayList<ConnectionPoint> connPointList;
017            /**
018            *List of Connections a Component has.
019            */
020            ArrayList<Connection> myConns;
021            /**
022            *Orientation of Component.
023            */
024            private Orientation orient;
025            /**
026            *Name of the component
027            */
028            private String name;
029            /**
030            *Location of the component
031            */
032            private Point location;
033            public CMComponent() {
034                    connPointList = new ArrayList<ConnectionPoint>();
035                    myConns = new ArrayList<Connection>();
036                    orient = Orientation.Up;
037                    name = "";
038                    location = new Point (0, 0);
039            }
040    
041            /**
042            * Connects this component to another
043            * @param a Component connection point to connect to.
044            * @param b Component connection point to connect to.
045            * @test.status ok
046            */
047            public void connectTo(ConnectionPoint a, ConnectionPoint b) {
048                    Connection conn = new CMConnection();
049                    conn.setEnds(a, b);
050                    myConns.add(conn);
051            }
052    
053            /**
054            * Rotates this component
055            * @param newOrient New orientation for Component.
056            * @test.status ok
057            */
058            public void rotate(Orientation newOrient) {
059                    orient = newOrient;
060            }
061    
062            /**
063            * Returns the orientation of the component (changed with rotate())
064            * @return Orientation enum describing current orientation: Up, Down, Left, Right (facing)
065            * @test.status ok
066            */
067            public Orientation getOrientation() {
068                    return orient;
069            }
070    
071            /**
072            * Returns the name of this component.
073            * Likely helpful for use in GUI, and debugging
074            * @return String name of component.
075            * @test.status ok
076            */
077            public String getName() {
078                    return name;
079            }
080    
081            /**
082            * Sets name of component.
083            * @param n New new for component.
084            * @test.status ok
085            */
086            public void setName(String n) {
087                    name = n;
088            }
089    
090            /**
091            * Returns a list of Connections this component has
092            * @return List of connections component has
093            * @test.status ok
094            */
095            public List<Connection> getConnections() {
096                    return myConns;
097            }
098    
099            /**
100            *Gets connection points for a component.
101            *@return list of ConnectionPoints 
102            *@test.status ok
103            */
104            public List<ConnectionPoint> getConnectionPoints() {
105                    return connPointList;
106            }
107    
108            /**
109            *Adds a connection point to a component
110            *@param newPoint ConnectionPoint to add to Component.
111            *@test.status ok
112            */
113            public void addConnectionPoint(ConnectionPoint newPoint) {
114                    connPointList.add(newPoint);
115            }
116    
117            /**
118            * Returns location of Component
119            * @return Point object with location coordinates
120            * @test.status ok
121            */
122            public Point getPosition() {
123                    return location;
124            }
125    
126            /**
127            * Sets the position
128            * @param x X coordinate for new location of component
129            * @param y Y coordination for new location of component
130            * @test.status ok
131            */
132            public void setPosition(int x, int y) {
133                    location.setLocation(x, y);
134            }
135            
136            public static void main(String[] sa) throws Exception {
137                    CMComponent myComp = new CMComponent();
138                    ConnectionPoint cp1 = new CMConnectionPoint(1, 1, 5, 5);
139                    myComp.rotate(Orientation.Left);
140                    myComp.setName("test comp 1");
141                    myComp.addConnectionPoint(cp1);
142                    myComp.setPosition(5, 6);
143                    CMComponent myComp2 = new CMComponent();
144                    ConnectionPoint cp2 = new CMConnectionPoint(3, 3, 5, 5);
145                    myComp2.addConnectionPoint(cp2);
146                    myComp.connectTo(cp1, cp2);
147                    System.out.println(myComp.getConnectionPoints());
148                    System.out.println(myComp.getConnections());
149                    System.out.println(myComp.getOrientation());
150            }
151    }
