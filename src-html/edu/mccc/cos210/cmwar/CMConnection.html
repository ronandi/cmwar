001    package edu.mccc.cos210.cmwar;
002    import java.util.List;
003    import java.util.ArrayList;
004    
005    /**
006    * CMConnection for Circuit Maker with AutoRouter
007    * A Connection is what can connect two Connectables / Components
008    * @author Rohit Kumar
009    * @version 0.1
010    */
011    public class CMConnection implements Connection {
012            /**
013            *Array of size 2 holding connection points
014            */
015            private ConnectionPoint[] ends;
016            public CMConnection() {
017                    ends = new ConnectionPoint[2];
018            }
019    
020            /**
021            * Method to get the components connected by this connection
022            * @return An array of size 2 with the ConnectionPoints
023            * @test.status ok
024            */
025            public ConnectionPoint[] getEnds() {
026                    return ends;
027            }
028    
029            /**
030            * Method to set the ends of a connection.
031            * @param a A Component to be connected.
032            * @param b Opposite Component to be connected to A, and vice versa.
033            * @test.status ok
034            */
035            public void setEnds(ConnectionPoint a, ConnectionPoint b) {
036                    ends[0] = a;
037                    ends[1] = b;
038            }
039            public static void main(String[] sa) throws Exception {
040                    ConnectionPoint cp1 = new CMConnectionPoint(1, 1, 5, 5);
041                    ConnectionPoint cp2 = new CMConnectionPoint(3, 3, 5, 5);
042                    CMConnection myConnection = new CMConnection();
043                    myConnection.setEnds(cp1, cp2);
044            }
045    }
