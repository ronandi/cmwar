001    package edu.mccc.cos210.cmwar;
002    import java.util.List;
003    /**
004    * Connection interface for Circuit Maker with AutoRouter
005    * A Connection is what can connect two Connectables / Components
006    * @author Rohit Kumar
007    * @version 0.1
008    */
009    public interface Connection {
010            /**
011            * Method to get the components connected by this connection
012            * @return An array of size 2 with the Components
013            */
014            public Component[] getEnds();
015    
016            /**
017            * Method to set the ends of a connection.
018            * @param a A Component to be connected.
019            * @param b Opposite Component to be connected to A, and vice versa.
020            */
021            public void setEnds(Component a, Component b);
022    }
